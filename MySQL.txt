CREATE TABLE `project` (
  `Project_ID` varchar(50) NOT NULL,
  `Term` varchar(30) DEFAULT NULL,
  `Year` int(4) DEFAULT NULL,
  `Project_No` varchar(100) DEFAULT NULL,
  `Title` varchar(400) DEFAULT NULL,
  `Description` varchar(2500) DEFAULT NULL,
  `Student_Skills` varchar(2500) DEFAULT NULL,
  `Student_Duties` varchar(2500) DEFAULT NULL,
  `Student_Benefits` varchar(2500) DEFAULT NULL,
  `Resources` varchar(2500) DEFAULT NULL,
  `Instructor_ID` varchar(50) DEFAULT NULL,
  `Sponsor_ID` varchar(50) DEFAULT NULL,
  PRIMARY KEY (`Project_ID`),
  KEY `Instructor_ID` (`Instructor_ID`),
  KEY `Sponsor_ID` (`Sponsor_ID`),
  CONSTRAINT `project_ibfk_1` FOREIGN KEY (`Instructor_ID`) REFERENCES `instructor` (`Instructor_ID`),
  CONSTRAINT `project_ibfk_2` FOREIGN KEY (`Sponsor_ID`) REFERENCES `project_sponsor` (`Sponsor_ID`)
)

CREATE TABLE `sponsor_update` (
  `Sponsor_Update_ID` varchar(50) NOT NULL,
  `Sponsor_ID` varchar(50) DEFAULT NULL,
  `project_ID` varchar(50) DEFAULT NULL,
  `Date` datetime DEFAULT NULL,
  `Progress` int(30) DEFAULT NULL,
  `Responsiveness` int(30) DEFAULT NULL,
  `Feedback` varchar(2500) DEFAULT NULL,
  PRIMARY KEY (`Sponsor_Update_ID`),
  UNIQUE KEY `project_ID` (`project_ID`),
  CONSTRAINT `sponsor_update_ibfk_1` FOREIGN KEY (`project_ID`) REFERENCES `project` (`Project_ID`)
)

CREATE TABLE `student_update` (
  `Student_Update_ID` varchar(50) NOT NULL,
  `Student_ID` varchar(50) DEFAULT NULL,
  `Project_ID` varchar(50) DEFAULT NULL,
  `Date` datetime DEFAULT NULL,
  `Progress` int(30) DEFAULT NULL,
  `Summary` varchar(2500) DEFAULT NULL,
  PRIMARY KEY (`Student_Update_ID`),
  KEY `Project_ID` (`Project_ID`),
  KEY `Student_ID` (`Student_ID`),
  CONSTRAINT `student_update_ibfk_1` FOREIGN KEY (`Project_ID`) REFERENCES `project` (`Project_ID`)
)
